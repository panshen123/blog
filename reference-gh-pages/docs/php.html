Nullsafe 运算符

// 从 PHP 8.0.0 开始，这一行：
$result = $repo?->getUser(5)?->name;
// 相当于下面的代码：
if (is_null($repo)) {
    $result = null;
} else {
    $user = $repository->getUser(5);
    if (is_null($user)) {
        $result = null;
    } else {
        $result = $user->name;
    }
}
